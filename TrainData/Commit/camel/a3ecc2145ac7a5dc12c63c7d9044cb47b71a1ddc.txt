	private static String loadJsonSchema(ClassLoader classLoader, String jsonSchemaPath) {
		try (InputStream is = classLoader.getResourceAsStream(jsonSchemaPath)) {
            if (is != null) {
                return loadText(is);
            }
        } catch (Throwable e) {
            LOG.warn("Error loading "  jsonSchemaPath  " file", e);
        }
		return null;
	}
