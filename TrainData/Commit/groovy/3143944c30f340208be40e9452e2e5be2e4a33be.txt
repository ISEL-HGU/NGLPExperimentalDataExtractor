        boolean tmp;
        try {
            tmp = TRUE_STR.equals(System.getProperty("groovy.enable.parameterized.type.cache", TRUE_STR));
        } catch (Exception e) {
            tmp = true;
        }

        PARAMETERIZED_TYPE_CACHE_ENABLED = tmp;
    }
