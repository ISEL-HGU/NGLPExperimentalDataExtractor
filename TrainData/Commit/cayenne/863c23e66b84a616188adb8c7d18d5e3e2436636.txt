    public void testPageSizeOneScalarAsArray() {
        List<Object[]> a = ObjectSelect.query(Artist.class)
                .columns(Artist.ARTIST_NAME.trim())
                .pageSize(10)
                .select(context);
        assertNotNull(a);

        }
