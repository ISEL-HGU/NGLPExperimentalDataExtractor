	private boolean hasDisjointNode(PrefetchTreeNode prefetchElement) {
		if (prefetchElement.isDisjointPrefetch()) {
			return true;
		}
		for (PrefetchTreeNode child : prefetchElement.getChildren()) {
			if (hasDisjointNode(child)) {
				return true;
			}
		}
		return false;
	}
